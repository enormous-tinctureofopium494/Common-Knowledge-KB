-- =============================================================
-- Common Knowledge KB — Interactive Examples (v0.5)
-- =============================================================
-- Compatibility notes vs. v3 examples:
--
--   KIND / UUID SCHEME:
--   • All named individuals, roles, and type-objects now use
--     kind = 'entity' and stable_uuid(name, 'entity').
--   • Exceptions: sources use 'source', contexts use 'context',
--     predicates use 'predicate'.  These are unchanged.
--   • Named individuals (ada_lovelace, charles_babbage, …) also
--     use 'entity'; there is no longer a 'person' or 'institution'
--     kind.  Type information is expressed via is_a statements and
--     type_membership rows.
--
--   NEGATION:
--   • negated boolean removed from statements (v0.5).
--   • Evidential opposition is expressed by inserting a competing
--     statement and registering a conflicts row with
--     conflict_kind = 'direct_negation'.
--   • Part G is rewritten accordingly.
--
--   COLUMN RENAMES:
--   • statement_belief.mean → belief_mean (stored generated column).
--   • predicates.domains dropped; queries updated.
--
--   holds_at() SIGNATURE:
--   • Now takes p_mode text DEFAULT 'open_world'.
--     Examples pass p_mode explicitly for clarity.
--   • Default mode 'open_world' returns only belief_mean > 0.5.
--     Use 'evidence_weighted' to get all matching statements.
--
--   update_trust() SIGNATURE:
--   • Now takes p_override boolean DEFAULT false.
--     Examples pass it explicitly.
--
--   OBJECT ARGS KEY ALIGNMENT:
--   • All stable_uuid() calls for type objects (person, mammal,
--     concept_type, …) updated to use 'entity' suffix to match
--     ckb_basis_objects_v0.5.sql.
--
-- =============================================================


-- ════════════════════════════════════════════════════════════
-- PART A: INSPECTION
-- ════════════════════════════════════════════════════════════

-- ── A1. Object counts by kind ────────────────────────────────

SELECT kind, count(*) AS n
FROM objects
GROUP BY kind
ORDER BY n DESC;


-- ── A2. All basis predicates — arity, transitivity, strictness ─

SELECT
    o.canonical_name            AS predicate,
    p.arity,
    CASE p.arity
        WHEN 1 THEN 'intransitive'
        WHEN 2 THEN 'transitive'
        WHEN 3 THEN 'bitransitive'
        ELSE        'higher-order'
    END                         AS transitivity,
    p.arg_labels,
    p.domain_strictness,
    p.nl_description
FROM predicates p
JOIN objects o ON o.id = p.id
WHERE p.is_basis = true
ORDER BY p.arity, o.canonical_name;


-- ── A3. Full type hierarchy with belief scores ───────────────

SELECT
    child.canonical_name              AS subtype,
    parent.canonical_name             AS supertype,
    round(sb.belief_mean::numeric, 4) AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb  ON sb.id     = s.id
JOIN objects child        ON child.id  = s.object_args[1]
JOIN objects parent       ON parent.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('subtype_of', 'predicate')
ORDER BY parent.canonical_name, child.canonical_name;


-- ── A4. All disjointness axioms ──────────────────────────────

SELECT
    a.canonical_name              AS type_a,
    b.canonical_name              AS type_b,
    round(sb.belief_mean::numeric, 4) AS belief
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects a           ON a.id   = s.object_args[1]
JOIN objects b           ON b.id   = s.object_args[2]
WHERE s.predicate_id = stable_uuid('disjoint_with', 'predicate')
ORDER BY type_a, type_b;


-- ── A5. Source credibility priors ────────────────────────────

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,
    round(scs.evidence_strength::numeric, 1)  AS evidence_strength,
    scs.is_protected
FROM source_credibility_score scs
JOIN objects o ON o.id = scs.source_id
JOIN objects c ON c.id = scs.context_id
ORDER BY scs.mean DESC;

-- ── B1. Boolean: "Is a mammal an animal?" ────────────────────

SELECT
    child.canonical_name              AS subtype,
    parent.canonical_name             AS supertype,
    round(sb.belief_mean::numeric, 4) AS belief_mean,
    round(sb.ci_low::numeric, 4)      AS ci_low,
    round(sb.ci_high::numeric, 4)     AS ci_high,
    sb.evidence_strength,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name  = 'mammal'
  AND parent.canonical_name = 'animal';
-- Expect: belief_mean ≈ 1.0000, t_kind = 'eternal'


-- ── B2. Binding: "What is person a subtype of?" ──────────────

SELECT
    parent.canonical_name             AS is_subtype_of,
    round(sb.belief_mean::numeric, 4) AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id       = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name = 'person'
ORDER BY sb.belief_mean DESC;
-- Expect: mammal, agent (both near-certain, eternal)


-- ── B3. Reverse binding: "What are direct subtypes of entity?" ─

SELECT
    child.canonical_name              AS subtype,
    round(sb.belief_mean::numeric, 4) AS belief
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND parent.canonical_name = 'entity'
ORDER BY sb.belief_mean DESC;


-- ── B4. Typicality: all typical_of statements ranked ─────────

SELECT
    instance.canonical_name           AS instance,
    type_obj.canonical_name           AS typical_of,
    round(sb.belief_mean::numeric, 4) AS typicality,
    round(sb.ci_low::numeric, 4)      AS ci_low,
    round(sb.ci_high::numeric, 4)     AS ci_high
FROM statements s
JOIN statement_belief sb ON sb.id     = s.id
JOIN objects instance    ON instance.id = s.object_args[1]
JOIN objects type_obj    ON type_obj.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('typical_of', 'predicate')
ORDER BY sb.belief_mean DESC;
-- Expect: person→agent highest (~0.90); institution→agent lower (~0.60)

-- ── F1. Forward-chain: "Lovelace is a mammal" ────────────────
-- Chain: lovelace is_a person ∧ person subtype_of mammal
--        → lovelace is_a mammal
-- In v0.5, provenance is recorded in statement_dependencies
-- (authoritative) and derived_from uuid[] (cache).

DO $$
DECLARE
    ada_id     uuid := stable_uuid('ada_lovelace', 'entity');
    mammal_id  uuid := stable_uuid('mammal',       'entity');
    person_id  uuid := stable_uuid('person',       'entity');
    is_a_id    uuid := stable_uuid('is_a',         'predicate');
    subtype_id uuid := stable_uuid('subtype_of',   'predicate');
    reality_id uuid := stable_uuid('reality',      'context');
    src_stmt_1 uuid;
    src_stmt_2 uuid;
    new_stmt   uuid;
BEGIN
    SELECT id INTO src_stmt_1
    FROM statements
    WHERE predicate_id = is_a_id
      AND object_args  = ARRAY[ada_id, person_id]
    LIMIT 1;

    SELECT id INTO src_stmt_2
    FROM statements
    WHERE predicate_id = subtype_id
      AND object_args  = ARRAY[person_id, mammal_id]
    LIMIT 1;

    IF src_stmt_1 IS NULL OR src_stmt_2 IS NULL THEN
        RAISE NOTICE 'Source statements not found; skipping derivation.';
        RETURN;
    END IF;

    INSERT INTO statements (
        predicate_id, object_args,
        belief_alpha, belief_beta,
        t_kind, context_id,
        derivation_type, derivation_depth, derived_from
    ) VALUES (
        is_a_id,
        ARRAY[ada_id, mammal_id],
        950.0, 0.001,
        'eternal', reality_id,
        'forward_chained', 1,
        ARRAY[src_stmt_1, src_stmt_2]
    )
    RETURNING id INTO new_stmt;

    -- Record in the authoritative dependency graph
    INSERT INTO statement_dependencies (parent_id, child_id, rule_name, weight)
    VALUES
        (src_stmt_1, new_stmt, 'is_a + subtype_of → is_a', 1.0),
        (src_stmt_2, new_stmt, 'is_a + subtype_of → is_a', 1.0);

    RAISE NOTICE 'Derived statement created: % (lovelace is_a mammal)', new_stmt;
END $$;

-- ── G2. Detect the conflict ───────────────────────────────────
-- Find statement pairs with the same predicate and args where
-- belief_means are in strong opposition (one high, one low).
-- This is how the conflict detector identifies direct negation
-- candidates without a negated boolean.

WITH role_stmts AS (
    SELECT
        s.id,
        s.object_args,
        sb.belief_mean AS belief
    FROM statements s
    JOIN statement_belief sb ON sb.id = s.id
    WHERE s.predicate_id = stable_uuid('has_role', 'predicate')
)
SELECT
    a.id                              AS stmt_high_belief,
    b.id                              AS stmt_low_belief,
    a.object_args,
    round(a.belief::numeric, 4)       AS high_belief,
    round(b.belief::numeric, 4)       AS low_belief,
    round((a.belief - b.belief)::numeric, 4) AS belief_gap
FROM role_stmts a
JOIN role_stmts b
    ON  a.object_args = b.object_args
    AND a.belief      > 0.5           -- high-belief statement
    AND b.belief      < 0.5           -- low-belief opposing statement
    AND a.id          <> b.id
    AND a.id          < b.id;         -- canonical pair ordering
-- Expect: one row.
-- belief_gap ≈ 0.59: strong positive claim (≈0.88) vs weak opposing (≈0.29).
-- The Bayesian model correctly reflects the credibility asymmetry.


-- ── G3. Register the conflict formally ───────────────────────

DO $$
DECLARE
    high_id uuid;
    low_id  uuid;
BEGIN
    -- The high-belief statement (wikidata + biography)
    SELECT id INTO high_id
    FROM statements
    WHERE predicate_id   = stable_uuid('has_role', 'predicate')
      AND object_args[1] = stable_uuid('ada_lovelace',  'entity')
      AND object_args[2] = stable_uuid('mathematician', 'entity')
    ORDER BY belief_mean DESC LIMIT 1;

    -- The low-belief opposing statement (unreliable blog)
    SELECT id INTO low_id
    FROM statements
    WHERE predicate_id   = stable_uuid('has_role', 'predicate')
      AND object_args[1] = stable_uuid('ada_lovelace',  'entity')
      AND object_args[2] = stable_uuid('mathematician', 'entity')
    ORDER BY belief_mean ASC LIMIT 1;

    IF high_id IS NOT NULL AND low_id IS NOT NULL AND high_id <> low_id THEN
        INSERT INTO conflicts (statement_a, statement_b, conflict_kind)
        VALUES (high_id, low_id, 'direct_negation')
        ON CONFLICT DO NOTHING;
        RAISE NOTICE 'Conflict registered between % (high) and % (low)',
            high_id, low_id;
    END IF;
END $$;


-- ── G4. Resolve a conflict ────────────────────────────────────
-- After review, the human or reasoning layer marks the conflict
-- resolved and records why.

UPDATE conflicts
SET resolved        = true,
    resolution_note = 'Unreliable blog (α=2, β=5, mean≈0.29) outweighed '
                      'by Wikidata + Lovelace biography (α≈14.9, β≈2, mean≈0.88). '
                      'High-belief statement retained.'
WHERE conflict_kind = 'direct_negation'
  AND statement_a IN (
      SELECT id FROM statements
      WHERE predicate_id   = stable_uuid('has_role', 'predicate')
        AND object_args[1] = stable_uuid('ada_lovelace',  'entity')
        AND object_args[2] = stable_uuid('mathematician', 'entity')
  );
