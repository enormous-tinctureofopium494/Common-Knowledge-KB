-- =============================================================
-- Common Knowledge KB — Interactive Examples  (v0.3)

-- Run these in psql or any Postgres client after loading:
--   1. common_knowledge_schema.sql
--   2. common_predicates_kernel.sql
--   3. common_objects_kernel.sql
--
-- Examples progress from inspection → assertion → temporal
-- queries → Bayesian updates → conflict detection → introspection.
--
-- Schema v3 notes:
--   • arg_types lives on predicates, not statements.
--   • has_role(entity, role, scope) replaces held_office.
--   • All UUIDs resolved via stable_uuid() — no hardcoded hex.
--   • update_trust() accepts an optional p_weight parameter.
--   • conflicts.conflict_kind is an enum, not free text.

-- ════════════════════════════════════════════════════════════
-- PART A: INSPECTION
-- ════════════════════════════════════════════════════════════

-- ── A1. Object counts by kind ────────────────────────────────

SELECT kind, count(*) AS n
FROM objects
GROUP BY kind
ORDER BY n DESC;


-- ── A2. All basis predicates — arity, transitivity, domains ──

SELECT
    o.canonical_name                        AS predicate,
    p.arity,
    CASE p.arity
        WHEN 1 THEN 'intransitive'
        WHEN 2 THEN 'transitive'
        WHEN 3 THEN 'bitransitive'
        ELSE        'higher-order'
    END                                     AS transitivity,
    p.arg_labels,
    p.domains,
    p.nl_description
FROM predicates p
JOIN objects o ON o.id = p.id
WHERE p.is_basis = true
ORDER BY p.arity, p.domains[1], o.canonical_name;


-- ── A3. Full type hierarchy with belief scores ───────────────

SELECT
    child.canonical_name        AS subtype,
    parent.canonical_name       AS supertype,
    round(sb.mean::numeric, 4)  AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb  ON sb.id  = s.id
JOIN objects child        ON child.id  = s.object_args[1]
JOIN objects parent       ON parent.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('subtype_of', 'predicate')
ORDER BY parent.canonical_name, child.canonical_name;


-- ── A4. All disjointness axioms ──────────────────────────────
-- These drive type-violation conflict detection.

SELECT
    a.canonical_name            AS type_a,
    b.canonical_name            AS type_b,
    round(sb.mean::numeric, 4)  AS belief
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects a           ON a.id   = s.object_args[1]
JOIN objects b           ON b.id   = s.object_args[2]
WHERE s.predicate_id = stable_uuid('disjoint_with', 'predicate')
ORDER BY type_a, type_b;


-- ── A5. Source credibility priors ────────────────────────────

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,
    round(scs.evidence_strength::numeric, 1)  AS evidence_strength
FROM source_credibility_score scs
JOIN objects o ON o.id = scs.source_id
JOIN objects c ON c.id = scs.context_id
ORDER BY scs.mean DESC;

-- PART B: SIMPLE QUERIES ON KERNEL CONTENT
-- ════════════════════════════════════════════════════════════

-- ── B1. Boolean: "Is a mammal an animal?" ────────────────────

SELECT
    child.canonical_name          AS subtype,
    parent.canonical_name         AS supertype,
    round(sb.mean::numeric, 4)    AS belief_mean,
    round(sb.ci_low::numeric, 4)  AS ci_low,
    round(sb.ci_high::numeric, 4) AS ci_high,
    sb.evidence_strength,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name  = 'mammal'
  AND parent.canonical_name = 'animal';
-- Expect: belief_mean ≈ 1.0000, t_kind = 'eternal'


-- ── B2. Binding: "What is person a subtype of?" ──────────────

SELECT
    parent.canonical_name         AS is_subtype_of,
    round(sb.mean::numeric, 4)    AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id       = stable_uuid('subtype_of', 'predicate')
  AND child.canonical_name = 'person'
ORDER BY sb.mean DESC;
-- Expect: mammal, agent (both near-certain, eternal)


-- ── B3. Reverse binding: "What are direct subtypes of entity?" ─

SELECT
    child.canonical_name          AS subtype,
    round(sb.mean::numeric, 4)    AS belief
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects child       ON child.id  = s.object_args[1]
JOIN objects parent      ON parent.id = s.object_args[2]
WHERE s.predicate_id        = stable_uuid('subtype_of', 'predicate')
  AND parent.canonical_name = 'entity'
ORDER BY sb.mean DESC;


-- ── B4. Typicality: all typical_of statements ranked ─────────

SELECT
    instance.canonical_name       AS instance,
    type_obj.canonical_name       AS typical_of,
    round(sb.mean::numeric, 4)    AS typicality,
    round(sb.ci_low::numeric, 4)  AS ci_low,
    round(sb.ci_high::numeric, 4) AS ci_high
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects instance    ON instance.id = s.object_args[1]
JOIN objects type_obj    ON type_obj.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('typical_of', 'predicate')
ORDER BY sb.mean DESC;
-- Expect: person→agent highest (~0.90); institution→agent lower (~0.60)


-- ── B5. Check a disjointness: "Are numbers and organisms disjoint?" ─

SELECT
    a.canonical_name              AS type_a,
    b.canonical_name              AS type_b,
    round(sb.mean::numeric, 4)    AS belief,
    s.t_kind
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects a           ON a.id   = s.object_args[1]
JOIN objects b           ON b.id   = s.object_args[2]
WHERE s.predicate_id       = stable_uuid('disjoint_with', 'predicate')
  AND a.canonical_name     = 'number'
  AND b.canonical_name     = 'organism';
-- Expect: belief_mean ≈ 1.0000


-- ── B6. Check person-institution disjointness (graded) ───────

SELECT
    a.canonical_name              AS type_a,
    b.canonical_name              AS type_b,
    round(sb.mean::numeric, 4)    AS belief,
    round(sb.ci_low::numeric, 4)  AS ci_low
FROM statements s
JOIN statement_belief sb ON sb.id  = s.id
JOIN objects a           ON a.id   = s.object_args[1]
JOIN objects b           ON b.id   = s.object_args[2]
WHERE s.predicate_id       = stable_uuid('disjoint_with', 'predicate')
  AND a.canonical_name     = 'person'
  AND b.canonical_name     = 'institution';
-- Expect: belief ≈ 0.90 (not near-certain: sole-trader edge case)

-- Check belief before update
SELECT
    round(sb.mean::numeric, 4)            AS belief_before,
    round(sb.evidence_strength::numeric,2) AS evidence_strength
FROM statements s
JOIN statement_belief sb ON sb.id = s.id
WHERE s.predicate_id = stable_uuid('has_role', 'predicate')
  AND s.object_args[1] = stable_uuid('ada_lovelace',  'person')
  AND s.object_args[2] = stable_uuid('mathematician', 'concept')
ORDER BY s.created_at DESC LIMIT 1;

-- Now update belief: biography credibility mean in history = 0.90
SELECT update_belief(
    (SELECT id FROM statements
     WHERE predicate_id = stable_uuid('has_role', 'predicate')
       AND object_args[1] = stable_uuid('ada_lovelace',  'person')
       AND object_args[2] = stable_uuid('mathematician', 'concept')
     ORDER BY created_at DESC LIMIT 1),
    0.90,   -- lovelace_biography credibility mean in domain_history
    true    -- supports the claim
);

-- Check belief after update
SELECT
    round(sb.mean::numeric, 4)            AS belief_after,
    round(sb.ci_low::numeric, 4)          AS ci_low,
    round(sb.ci_high::numeric, 4)         AS ci_high,
    round(sb.evidence_strength::numeric,2) AS evidence_strength,
    round(sb.variance::numeric, 6)        AS variance
FROM statements s
JOIN statement_belief sb ON sb.id = s.id
WHERE s.predicate_id = stable_uuid('has_role', 'predicate')
  AND s.object_args[1] = stable_uuid('ada_lovelace',  'person')
  AND s.object_args[2] = stable_uuid('mathematician', 'concept')
ORDER BY s.created_at DESC LIMIT 1;
-- mean rises from ~0.867 to ~0.879.
-- Small update: prior evidence_strength was already 15; new evidence adds 0.90.
-- This is correct Bayesian behaviour — high prior evidence resists easy revision.

-- ── F1. Forward-chain: "Lovelace is a mammal" ────────────────
-- Chain: lovelace is_a person → person subtype_of mammal
--        → lovelace is_a mammal
-- This is a forward-chained statement derived from two source statements.

DO $$
DECLARE
    ada_id     uuid := stable_uuid('ada_lovelace', 'person');
    mammal_id  uuid := stable_uuid('mammal',       'concept');
    is_a_id    uuid := stable_uuid('is_a',         'predicate');
    subtype_id uuid := stable_uuid('subtype_of',   'predicate');
    reality_id uuid := stable_uuid('reality',      'context');
    person_id  uuid := stable_uuid('person',       'concept');
    src_stmt_1 uuid;
    src_stmt_2 uuid;
    new_stmt   uuid;
BEGIN
    -- Source statement 1: lovelace is_a person
    SELECT id INTO src_stmt_1
    FROM statements
    WHERE predicate_id = is_a_id
      AND object_args  = ARRAY[ada_id, person_id]
    LIMIT 1;

    -- Source statement 2: person subtype_of mammal
    SELECT id INTO src_stmt_2
    FROM statements
    WHERE predicate_id = subtype_id
      AND object_args  = ARRAY[person_id, mammal_id]
    LIMIT 1;

    IF src_stmt_1 IS NULL OR src_stmt_2 IS NULL THEN
        RAISE NOTICE 'Source statements not found; skipping derivation.';
        RETURN;
    END IF;

    INSERT INTO statements (
        predicate_id, object_args,
        belief_alpha, belief_beta,
        t_kind, context_id,
        derivation_type, derivation_depth, derived_from
    ) VALUES (
        is_a_id,
        ARRAY[ada_id, mammal_id],
        -- Belief: min of source beliefs, slightly discounted for chain length
        950.0, 0.001,
        'eternal', reality_id,
        'forward_chained', 1,
        ARRAY[src_stmt_1, src_stmt_2]
    )
    RETURNING id INTO new_stmt;

    RAISE NOTICE 'Derived statement created: % (lovelace is_a mammal)', new_stmt;
END $$;


-- ── F2. Provenance query: all statements about Lovelace ───────

SELECT
    p.canonical_name                                        AS predicate,
    sv.arg_names                                            AS args,
    round(sv.belief_mean::numeric, 4)                       AS belief,
    sv.evidence_strength,
    sv.t_kind,
    sv.derivation_type,
    sv.derivation_depth,
    sv.derived_from
FROM statement_view sv
JOIN objects p ON p.canonical_name = sv.predicate
WHERE stable_uuid('ada_lovelace', 'person') = ANY(sv.object_args)
ORDER BY sv.belief_mean DESC, sv.derivation_depth;


-- ── F3. Provenance query: what sources attest a specific claim ─

SELECT
    o_pred.canonical_name                           AS predicate,
    round(sb.mean::numeric, 4)                      AS belief_mean,
    s.t_kind,
    s.derivation_type,
    s.derivation_depth,
    array_agg(o_src.canonical_name ORDER BY a.created_at) AS sources,
    array_agg(a.confidence_weight  ORDER BY a.created_at) AS weights,
    array_agg(a.evidence_group_id  ORDER BY a.created_at) AS evidence_groups
FROM statements s
JOIN statement_belief sb  ON sb.id     = s.id
JOIN objects o_pred       ON o_pred.id = s.predicate_id
LEFT JOIN attestations a  ON a.statement_id = s.id
LEFT JOIN objects o_src   ON o_src.id  = a.source_id
WHERE s.predicate_id  = stable_uuid('has_role', 'predicate')
  AND s.object_args[1] = stable_uuid('ada_lovelace',  'person')
  AND s.object_args[2] = stable_uuid('mathematician', 'concept')
GROUP BY s.id, o_pred.canonical_name, sb.mean,
         s.t_kind, s.derivation_type, s.derivation_depth;

-- ── G2. Detect the conflict ───────────────────────────────────
-- Find pairs of statements with the same predicate and args where
-- one is negated and the other is not.

WITH role_stmts AS (
    SELECT
        s.id,
        s.object_args,
        s.negated,
        sb.mean AS belief
    FROM statements s
    JOIN statement_belief sb ON sb.id = s.id
    WHERE s.predicate_id = stable_uuid('has_role', 'predicate')
)
SELECT
    a.id                       AS stmt_positive,
    b.id                       AS stmt_negated,
    a.object_args,
    round(a.belief::numeric,4) AS positive_belief,
    round(b.belief::numeric,4) AS negative_belief,
    round((a.belief - b.belief)::numeric, 4) AS belief_gap
FROM role_stmts a
JOIN role_stmts b
    ON  a.object_args  = b.object_args
    AND a.negated      = false
    AND b.negated      = true
    AND a.id           < b.id;
-- Expect: one conflict row.
-- belief_gap ≈ 0.59: positive claim (mean≈0.88) vs negative claim (mean≈0.29).
-- The Bayesian model correctly reflects that the negative claim has far less
-- credible support — both statements exist, but the KB doesn't flip.


-- ── G3. Register the conflict formally ───────────────────────
-- Insert into the conflicts table with the correct conflict_kind enum value.

DO $$
DECLARE
    pos_id uuid;
    neg_id uuid;
BEGIN
    SELECT id INTO pos_id
    FROM statements
    WHERE predicate_id   = stable_uuid('has_role', 'predicate')
      AND object_args[1] = stable_uuid('ada_lovelace',  'person')
      AND object_args[2] = stable_uuid('mathematician', 'concept')
      AND negated        = false
    ORDER BY belief_alpha DESC LIMIT 1;

    SELECT id INTO neg_id
    FROM statements
    WHERE predicate_id   = stable_uuid('has_role', 'predicate')
      AND object_args[1] = stable_uuid('ada_lovelace',  'person')
      AND object_args[2] = stable_uuid('mathematician', 'concept')
      AND negated        = true
    ORDER BY created_at DESC LIMIT 1;

    IF pos_id IS NOT NULL AND neg_id IS NOT NULL THEN
        INSERT INTO conflicts (statement_a, statement_b, conflict_kind)
        VALUES (pos_id, neg_id, 'direct_negation')
        ON CONFLICT DO NOTHING;
        RAISE NOTICE 'Conflict registered between % and %', pos_id, neg_id;
    END IF;
END $$;


-- ── G4. Type-violation conflict example ──────────────────────
-- A hypothetical: something claims the Difference Engine (an artifact)
-- is_a number — violates disjoint_with(physical_thing, abstract_thing).
-- We detect this via the disjointness axioms.

SELECT
    a.canonical_name        AS type_a,
    b.canonical_name        AS type_b,
    round(sb.mean::numeric,4) AS disjoint_belief
FROM statements s
JOIN statement_belief sb ON sb.id = s.id
JOIN objects a ON a.id = s.object_args[1]
JOIN objects b ON b.id = s.object_args[2]
WHERE s.predicate_id = stable_uuid('disjoint_with', 'predicate')
  AND (
    (a.canonical_name = 'physical_thing' AND b.canonical_name = 'abstract_thing')
    OR
    (a.canonical_name = 'abstract_thing' AND b.canonical_name = 'physical_thing')
  );

-- ── H1. Statement counts by t_kind and derivation_type ───────

SELECT
    t_kind,
    derivation_type,
    count(*)                                       AS n_statements,
    round(avg(sb.mean)::numeric, 4)                AS avg_belief,
    round(avg(sb.evidence_strength)::numeric, 2)   AS avg_evidence
FROM statements s

GROUP BY t_kind, derivation_type
ORDER BY t_kind, derivation_type;


-- ── H2. Introspect: "What does the KB believe about person?" ──
-- All statements in which 'person' appears as any object_arg.

SELECT
    p.canonical_name                                         AS predicate,
    sv.arg_names                                             AS args,
    round(sv.belief_mean::numeric, 4)                        AS belief,
    sv.t_kind,
    sv.derivation_type
FROM statement_view sv
JOIN objects p ON p.canonical_name = sv.predicate
WHERE stable_uuid('person', 'concept') = ANY(sv.object_args)
ORDER BY sv.belief_mean DESC;


-- ── H3. Introspect: "What does the KB know about the Difference Engine?" ─

SELECT
    p.canonical_name                                         AS predicate,
    sv.arg_names                                             AS args,
    round(sv.belief_mean::numeric, 4)                        AS belief,
    sv.t_kind
FROM statement_view sv
JOIN objects p ON p.canonical_name = sv.predicate
WHERE stable_uuid('difference_engine', 'concept') = ANY(sv.object_args)
ORDER BY sv.belief_mean DESC;


-- ── H4. Source credibility overview (all sources, all contexts) ─

SELECT
    o.canonical_name                          AS source,
    c.canonical_name                          AS context,
    round(scs.mean::numeric, 4)               AS credibility_mean,
    round(scs.ci_low::numeric, 4)             AS ci_low,
    round(scs.ci_high::numeric, 4)            AS ci_high,
    round(scs.evidence_strength::numeric, 1)  AS evidence_strength
FROM source_credibility_score scs
JOIN objects o ON o.id = scs.source_id
JOIN objects c ON c.id = scs.context_id
ORDER BY scs.mean DESC, o.canonical_name;


-- ── H5. Open conflicts ────────────────────────────────────────

SELECT
    c.id,
    c.conflict_kind,
    pa.canonical_name   AS predicate_a,
    pb.canonical_name   AS predicate_b,
    round(sba.mean::numeric,4) AS belief_a,
    round(sbb.mean::numeric,4) AS belief_b,
    c.resolved,
    c.created_at
FROM conflicts c
JOIN statements sa ON sa.id = c.statement_a
JOIN statements sb_s ON sb_s.id = c.statement_b
JOIN objects pa ON pa.id = sa.predicate_id
JOIN objects pb ON pb.id = sb_s.predicate_id
JOIN statement_belief sba ON sba.id = c.statement_a
JOIN statement_belief sbb ON sbb.id = c.statement_b
WHERE c.resolved = false
ORDER BY c.created_at;
